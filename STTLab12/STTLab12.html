class Subscriber

{

    public static void Ring(string t)

    {

        string currTime = DateTime.Now.TimeOfDay.ToString().Substring(0, 8);

        Console.WriteLine("Alarm was set on " + t + "\nIt's " + currTime + " already!!");

        Thread.Sleep(1000);

    }

}

class Publisher

{

    public event MyDel RaiseEvent;

    public delegate void MyDel(string t);

    public void SetTime(string sTime)

    {

        string currTime = DateTime.Now.TimeOfDay.ToString().Substring(0,8);

        while (currTime.CompareTo(sTime) <0 )

        {

            Console.WriteLine("Current Time:" + currTime);

            Thread.Sleep(1000);

            currTime = DateTime.Now.TimeOfDay.ToString().Substring(0,8);

        }

        RaiseEvent(currTime);

    }

    static void Main(string[] args)

    {

        Publisher p = new Publisher();

        p.RaiseEvent += new MyDel(Subscriber.Ring);

        Console.WriteLine("Type in time to set alarm in 24 hour format.");

        Console.WriteLine("Hours (two digits) :");

        string h = Console.ReadLine();

        while(h.Length != 2 || !(h.All(char.IsNumber)))

        {

            Console.WriteLine("Please enter 2 digits only. Try again");

            h = Console.ReadLine();

        }

        Console.WriteLine("Minutes (two digits) :");

        string m = Console.ReadLine();

        while (m.Length != 2 || !(m.All(char.IsNumber)))

        {

            Console.WriteLine("Please enter 2 digits only. Try again");

            m = Console.ReadLine();

        }

        Console.WriteLine("Seconds (two digits) :");

        string s = Console.ReadLine();

        while (s.Length != 2 || !(s.All(char.IsNumber)))

        {

            Console.WriteLine("Please enter 2 digits only. Try again");

            s = Console.ReadLine();

        }

        string t= h + ":"+ m + ":" + s;

        Console.WriteLine("Alarm set for " + t);

        p.SetTime(t);

    }

}

namespace PubSubForm

{

    public partial class Form1 : Form

    {

        private System.Windows.Forms.Timer timer; // a timer

        private string targetTime; // will be set on SetTime, just as in the console application

        private Random random = new Random(); // for selecting colors at random

        private bool alarmTriggered = false; // for control flow

        public Form1()

        {

            InitializeComponent();

            // Initialize the timer

            timer = new System.Windows.Forms.Timer();

            timer.Interval = 1000; // in ms

            timer.Tick += Timer_Tick; // handler

        }

        /* This is a handler for the timer.tick event.

         * We need to use this instead of Thread.sleep, because

         * Thread.Sleep will pause the main thread, which runs the GUI and handles events.

         */

        private void Timer_Tick(object sender, EventArgs e)

        {

            if (alarmTriggered)

                return;

            string currentTime = DateTime.Now.TimeOfDay.ToString().Substring(0, 8);

            // Change background color every second

            int r = random.Next(256);

            int g = random.Next(256);

            int b = random.Next(256);

            BackColor = Color.FromArgb(r, g, b); // BackColor attribute is used to set back-ground color of an object,

                                                 // in this case the Form1 object

                                                 // So, we can even skip the `this` as we have been doing for all attributes.

            // Check if target time is reached

            if (currentTime.CompareTo(targetTime) >= 0)

            {

                // Stop the timer

                timer.Stop();

                alarmTriggered = true;

                // Raise the alarm event

                Ring(targetTime);

                //RaiseAlarm?.Invoke(targetTime);

            }

        }

        public void Ring(string t)

        {

            string currTime = DateTime.Now.TimeOfDay.ToString().Substring(0, 8);

            // Display message box, as mentioned in the task

            MessageBox.Show("Alarm was set on " + t + "\nIt's " + currTime + " already!!", "Alarm",

                MessageBoxButtons.OK, MessageBoxIcon.Information);

            // Reset the form

            alarmTriggered = false;

            BackColor = SystemColors.Control; // Reset to default color

            label5.Text = "no alarm set";

        }

        private void button1_Click(object sender, EventArgs e)

        {

            string h = textBox1.Text;

            if (h.Length != 2 || !(h.All(char.IsDigit)))

            {

                label5.Text = "Please enter 2 digits only for hours";

                return;

            }

            string m = textBox2.Text;

            if (m.Length != 2 || !(m.All(char.IsDigit)))

            {

                label5.Text = "Please enter 2 digits only for minutes";

                return;

            }

            string s = textBox3.Text;

            if (s.Length != 2 || !(s.All(char.IsDigit)))

            {

                label5.Text = "Please enter 2 digits only for seconds";

                return;

            }

            // Validate time values

            if (int.Parse(h) > 23 || int.Parse(m) > 59 || int.Parse(s) > 59)

            {

                label5.Text = "Invalid time format. Please enter valid HH:MM:SS";

                return;

            }

            targetTime = h + ":" + m + ":" + s;

            label5.Text = "Alarm set for " + targetTime;

            // Reset alarm state

            alarmTriggered = false;

            // Start the timer

            timer.Start();

        }

    }

}
